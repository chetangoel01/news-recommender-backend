5. How to Set Up Google Authentication

Setup Required: To use real Google authentication, you need to:
  1. Create a Google OAuth 2.0 client in the Google Cloud Console
  2. Set the GOOGLE_CLIENT_ID environment variable
  3. Configure authorized origins and redirect URIs
  4. Use the Google Sign-In JavaScript library in your frontend

Example Frontend Integration:

<script src="https://accounts.google.com/gsi/client"></script>
<div id="g_id_onload"
     data-client_id="YOUR_GOOGLE_CLIENT_ID"
     data-callback="handleCredentialResponse">
</div>
<div class="g_id_signin" data-type="standard"></div>

<script>
function handleCredentialResponse(response) {
    // Send the ID token to your backend
    fetch('/auth/google', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            id_token: response.credential
        })
    })
    .then(res => res.json())
    .then(data => {
        // Handle successful authentication
        console.log('Authenticated:', data);
    })
    .catch(error => {
        console.error('Auth error:', error);
    });
}
</script>
